#include <iostream>
using namespace std;
                     
int main()
{
    int n,m,co=0;                        
    cout<<"Enter n"<<endl;          // 20   5
    cin>>n;                         // 4    5   
    cout<<"Enter m"<<endl;
    cin>>m;
    
    while(n != 0) // 20 15 10 5 0
    {
        if(n != 0)
        {
            n=n-m; //20-5 15 10 5
            co++; // 4
        }
        else
        {
            break;
        }
    }
    cout<<co;
}
